// ============================================ // DASHBOARD - VERSÃO SIMPLES // ============================================ import { db, collection, query, orderBy, onSnapshot } from './firebase-config.js'; import { escapeHtml, formatDate, showToast, checkAuth } from './utils.js'; import { getCurrentUser, logout } from './auth.js'; let allCourses = []; let userProgress = {}; // Inicialização export async function initDashboard() { console.log("Dashboard iniciado"); if (!checkAuth()) return; const user = getCurrentUser(); if (!user || user.type !== 'colaborador') { window.location.href = 'login.html'; return; } document.getElementById('userNameDisplay').textContent = user.name; document.getElementById('userAvatar').textContent = user.name.charAt(0).toUpperCase(); document.getElementById('welcomeMessage').innerHTML = `Bem-vindo de volta, ${user.name}! 👋`; // Carregar progresso const saved = localStorage.getItem(`progress_${user.name}`); if (saved) userProgress = JSON.parse(saved); // Carregar formações await loadCourses(); // Configurar eventos setupEventListeners(); } // Carregar formações async function loadCourses() { const loadingDiv = document.getElementById('loading'); const coursesGrid = document.getElementById('coursesGrid'); loadingDiv.style.display = 'block'; coursesGrid.style.display = 'none'; try { const unsubscribe = onSnapshot( query(collection(db, 'formacoes'), orderBy('dataTimestamp', 'desc')), (snapshot) => { allCourses = []; snapshot.forEach((doc) => { allCourses.push({ id: doc.id, ...doc.data() }); }); console.log("Formações carregadas:", allCourses.length); updateStats(); renderCourses(); loadingDiv.style.display = 'none'; coursesGrid.style.display = 'grid'; }, (error) => { console.error("Erro:", error); loadingDiv.innerHTML = '❌ Erro ao carregar formações.'; } ); } catch(e) { console.error("Erro:", e); loadingDiv.innerHTML = '❌ Erro ao carregar formações.'; } } // Atualizar estatísticas function updateStats() { let completed = 0; let inProgress = 0; allCourses.forEach(curso => { if (userProgress[curso.id]?.completed) completed++; else if (userProgress[curso.id]?.modulesCompleted > 0) inProgress++; }); document.getElementById('coursesCompleted').textContent = completed; document.getElementById('coursesInProgress').textContent = inProgress; document.getElementById('certificatesCount').textContent = completed; } // Calcular progresso function getProgress(curso) { const p = userProgress[curso.id]; if (p?.completed) return 100; if (p?.modulesCompleted) { const total = curso.modulos?.length || 1; return Math.round((p.modulesCompleted / total) * 100); } return 0; } // Renderizar cursos function renderCourses() { const grid = document.getElementById('coursesGrid'); const searchTerm = document.getElementById('searchInput')?.value.toLowerCase() || ''; const filter = document.querySelector('.filter-btn.active')?.dataset.filter || 'all'; let filtered = allCourses.filter(curso => { if (searchTerm && !curso.nome?.toLowerCase().includes(searchTerm)) return false; const progress = getProgress(curso); if (filter === 'concluidas') return progress === 100; if (filter === 'em_andamento') return progress > 0 && progress < 100; return true; }); if (filtered.length === 0) { grid.innerHTML = `
📚

Nenhuma formação

Tente ajustar os filtros.

`; return; } grid.innerHTML = filtered.map(curso => { const progress = getProgress(curso); const completed = progress === 100; const btnText = completed ? '🎓 Ver Certificado' : (progress > 0 ? '▶ Continuar' : '📖 Iniciar'); const btnClass = completed ? 'btn-completed' : (progress > 0 ? 'btn-continue' : ''); return `
${escapeHtml(curso.duracao || '30 min')}
${escapeHtml(curso.nome)}
${escapeHtml((curso.descricao || '').substring(0, 100))}
📦 ${curso.modulos?.length || 0} módulos 📝 ${curso.perguntas?.length || 0} questões
${!completed ? `
${progress}%
` : ''}
${btnText}
`; }).join(''); } // Entrar na formação window.entrarFormacao = (id) => { localStorage.setItem('cursoAtualId', id); window.location.href = 'formacao_colaborador.html'; }; // Configurar eventos function setupEventListeners() { const searchInput = document.getElementById('searchInput'); if (searchInput) { searchInput.addEventListener('input', () => renderCourses()); } document.querySelectorAll('.filter-btn').forEach(btn => { btn.addEventListener('click', () => { document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active')); btn.classList.add('active'); renderCourses(); }); }); document.querySelector('.logout-btn')?.addEventListener('click', logout); } // Iniciar document.addEventListener('DOMContentLoaded', () => { initDashboard(); }); window.entrarFormacao = entrarFormacao;
